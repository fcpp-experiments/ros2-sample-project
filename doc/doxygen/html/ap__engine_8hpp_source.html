92  FUN real_t run_rank_node(ARGS, float percent_charge, node_type nt, goal_tuple_type const& g) { CODE
93  // compute distance to target
94  real_t dist = norm(node.position() - make_vec(common::get<goal_pos_end_x>(g), common::get<goal_pos_end_y>(g), 0));
95  real_t rank = (nt == node_type::ROBOT)
96  ? dist * (1.0 - percent_charge )
97  : INF;
98  // rank setted to INF if:
99  auto is_current_goal = [&]()->bool {return node.storage(node_process_goal{}) == common::get<goal_code>(g);};
100  auto is_goal_failed = [&]()->bool {return node.storage(node_ext_goal_status{}) == feedback::GoalStatus::FAILED;};
101  auto is_goal_running = [&]()->bool {return node.storage(node_ext_goal_status{}) == feedback::GoalStatus::RUNNING;};
102  auto is_docking = [&]()->bool {return node.storage(node_ext_docking_status{}) == feedback::DockStatus::RUNNING;};
103  auto is_process_selected = [&]()->bool {return node.storage(node_process_status{}) == ProcessingStatus::SELECTED;};
104  auto is_goal_priority_0 = [&]()->bool {return common::get<goal_priority>(g) == 0;};
105  auto exists_prioritised_goal = [&]()->bool {return check_if_other_prioritised_goal_exists(CALL, common::get<goal_code>(g));};
106  if (
107  // node is failed for current goal and not already selected for something
108  (is_goal_failed() && is_current_goal() && !is_process_selected()) ||
109  // or it's already selected or running for another goal
110  ((is_goal_running() || is_process_selected()) && !is_current_goal()) ||
111  // or if it's processing not prioritised goal, but there are other some prioritised goals
112  (is_goal_priority_0() && exists_prioritised_goal()) ||
113  // or it's returning to its initial position
114  is_docking()
115  ) {
116  rank = INF;
117  }
118  // rank setted to INF if percent charge of battery is 0.0
119  else if (percent_charge == 0.0) {
120  rank = INF;
121  }
122  // rank set to dist * 0.01 when battery is fully charged, otherwise the rank will be always 0:
123  // the case when percent_charge is 1.0 (100%) becomes like percent_charge equals to 0.99 (99%)
124  else if (percent_charge >= 1.0) {
125  rank = dist * 0.01;
126  }
127 
128  if (AP_ENGINE_DEBUG) {
129  std::cout << "Computed local rank from node " << node.uid << " and for goal " << common::get<goal_code>(g) << " is: " << rank << endl;
130  }
131  return rank;
132  }

384  FUN void manage_action_goal(ARGS, node_type nt, goal_tuple_type const& g, status* s, int n_round) { CODE
385 
386  add_goal_to_computing_map(CALL, g);
387  // make_tuple(real_t, device_t, int, make_tuple(device_t, int)):
388  // [0] -> rank of current leader
389  // [1] -> node uid of current leader
390  // [2] -> counter of how many rounds the leader is
391  old(CALL, make_tuple(INF, node.uid, 0), [&] (rank_data_type current_rank_tuple) {
392 
393  // retrieve data collected from previous round
394  real_t current_rank = get<0>(current_rank_tuple);
395  device_t current_leader = get<1>(current_rank_tuple);
396  int current_leader_for_round = int(get<2>(current_rank_tuple));
397 
398  // compute charge of battery in percent
399  float percent_charge = node.storage(node_battery_charge{})/100.0;
400 
401  if (nt == node_type::ROBOT) {//i'm robot
402 
403  // if i'm terminating the current goal, i have to terminate goal for all nodes
404  if (common::get<goal_code>(g) == node.storage(node_process_goal{}) && //i was running current goal in the process
405  common::get<goal_code>(g) == node.storage(node_ext_goal{}) && //the robot was running current goal
406  ProcessingStatus::TERMINATING == node.storage(node_process_status{})) { //but now i'm terminating
407  return ends_processed_goal(CALL, current_rank, current_leader_for_round, nt, g, s);
408  }
409 
410  // if battery is empty, then stop at current position
411  if (node.storage(node_ext_goal_status{}) == feedback::GoalStatus::RUNNING && //i'm reaching the goal
412  percent_charge <= 0.0) { //the battery is full discharged
413  battery_discharged_when_it_is_running(CALL, g, s);
414  }
415  }
416 
417  update_leader_counter(CALL, percent_charge, current_rank, current_leader, current_leader_for_round, nt, g, s);
418 
419  // compute rank for this node
420  real_t computed_rank = INF;
421  computed_rank = run_rank_node(CALL, percent_charge, nt, g);
422 
423 
424  if (nt == node_type::ROBOT && //i'm robot
425  current_rank != INF && //i have computed something
426  computed_rank != INF && //and also now i have computed something
427  node.storage(node_process_status{}) == ProcessingStatus::IDLE && //i'm IDLE, so ready to go!
428  percent_charge > 0.0 && //i have sufficient battery
429  node.uid == current_leader && //i'm the best!
430  current_leader_for_round >= 2) // for at least 2 round
431  {
432 
433  send_action_to_selected_node(CALL, current_leader, g);
434  }
435 
436  // blinking colors if not running
437  if (common::get<goal_code>(g) != node.storage(node_process_goal{}) &&
438  node.storage(node_process_status{}) != ProcessingStatus::SELECTED) {
439  blink_computing_color(CALL, n_round);
440  }
441 
442  // i'm the leader!
443  return make_tuple(
444  0.0,
445  node.uid,
446  current_leader_for_round
447  );
448  });
449  }

545  old(CALL, robot_phase::IDLE, [&](robot_phase ph){

548

550  if (RobotStatesMap.find(rname) != RobotStatesMap.end()) {
551  for(RobotStatus rs : RobotStatesMap[rname]) {
552  // add offset to match with background

626  }

637  FUN void manage_termination(ARGS, node_type nt, spawn_result_type& r) {
638  // if process was terminating and now it's terminated, we have to change state machine to IDLE
639  if (node.storage(node_process_status{}) == ProcessingStatus::TERMINATING) {
640  // if process has been terminated, it isn't in the result map of spawn
641  bool process_found = false;
642  for (auto const& x : r)
643  {
644  auto g = x.first;
645  if (common::get<goal_code>(g) == node.storage(node_process_goal{})){
646  process_found = true;
647  }
648  }
649  if (!process_found) {
650  std::cout << "Process with code " << node.storage(node_process_goal{}) << " not found, so move robot to IDLE" << endl;
651  node.storage(node_process_status{}) = ProcessingStatus::IDLE;
652  }
653  }
654 
655  // search on results if the computing processes has been terminated:
656  // - if it's terminated (or in other words, if the goal it's not in the "r" variable), we delete it from the map in the storage
657  std::vector<std::string> goals_to_remove = {};
658  for (auto const& x : node.storage(node_process_computing_goals{}))
659  {
660  auto g = x.second;
661  if (r.find(g) == r.end()){
662  // std::cout << "Remove process with code " << common::get<goal_code>(g) << endl;
663  goals_to_remove.push_back(common::get<goal_code>(g));
664  }
665  }
666  for (auto const& gc : goals_to_remove) {
667  remove_goal_from_computing_map(CALL, gc);
668  }
669  }
